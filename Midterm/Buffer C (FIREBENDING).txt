void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    vec2 uv = fragCoord.xy / iResolution.xy;
    vec3 composite = texture(iChannel0, uv).rgb;
		
    //kernel declarations
    arraySize;
    kernelSize;
    float kernel[arraySize];
    vec3 finalColor = vec3(0.0);

    //one dimensional kernal creation
    //loops through a for loop to create the kernel
    float sigma = SIGMA;
    float negate = 0.0;
    for (int j = 0; j <= kernelSize; ++j)
    {
        kernel[kernelSize+j] = kernel[kernelSize-j] = normpdf(float(j), sigma);
    }

    //for loop that loops through to get the normalization factor
    //assuming the gaussian blur has been clamped
    for (int j = 0; j < arraySize; ++j)
    {
        negate += kernel[j];
    }

    //for loop inside of another for loop that reads out the texture channels to build the first layer of the flames
    for (int i=-kernelSize; i <= kernelSize; ++i)
    {
        for (int j=-kernelSize; j <= kernelSize; ++j)
        {
            finalColor += kernel[kernelSize+j]*kernel[kernelSize+i]*texture(iChannel0, (fragCoord.xy+vec2(float(i),float(j))) / iResolution.xy).rgb;
			//finalColor += 0.5 + 0.5*cos(iTime+uv.xyx+vec3(0.0,1.0,2.0));
        }
    }

    finalColor /= negate*negate;

    //finalColor = c + finalColor * 0.3;
	

    fragColor = vec4(finalColor, 1.0);
}
