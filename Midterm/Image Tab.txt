// mainImage: process the current pixel (exactly one call per pixel)
//    fragColor: output final color for current pixel
//    fragCoord: input location of current pixel in image (in pixels)



void mainImage(out color4 fragColor, in sCoord fragCoord)
{
    // setup
    // test UV for input image
    sCoord uv = fragCoord / iChannelResolution[0].xy;
    
    // TESTING
    // set iChannel0 to 'Misc/Buffer A' and fetch sample
    
    vec4 original = texture(iChannel0, uv);
    
    //original add
    //fragColor = original;
    
    //second layer of the flames creation by the buffer
    vec3 newLayer = texture(iChannel1, fragCoord.xy / iResolution.xy).rgb;
    //fragColor = vec4(c, 1.0);
    //return;
    //kernel declaration
    
    float kernel[arraySize];
    vec3 finalColor = vec3(0.0);

    //one dimensional kernel creation
    //for loop to create the base of the kernel size
    float sigma = SIGMA;
    float negate = 0.0;
    for (int j = 0; j <= kernelSize; ++j)
    {
        kernel[kernelSize+j] = kernel[kernelSize-j] = normpdf(float(j), sigma);
    }

    //for loop to get the normalization factor
    //assuming gaussian blur has been clamped already
    for (int j = 0; j < arraySize; ++j)
    {
        negate += kernel[j];
    }

    //for loop within a for loop to create the next layer of the flame to display out to the image
    for (int i=-kernelSize; i <= kernelSize; ++i)
    {
        for (int j=-kernelSize; j <= kernelSize; ++j)
        {
            finalColor += kernel[kernelSize+j]*kernel[kernelSize+i]*texture(iChannel2, (fragCoord.xy+vec2(float(i),float(j))) / iResolution.xy).rgb;

        }
    }

    finalColor /= negate*negate;

    //first layer addition to the main image
    finalColor = newLayer + pow(finalColor, vec3(0.5)) * 0.5;

    //flame layer addition to the original rotating cube
    fragColor = vec4(finalColor, 1.0) + original;

    //flame tracing addition to the overall final image
    fragColor += texture(iChannel3, uv);
}
